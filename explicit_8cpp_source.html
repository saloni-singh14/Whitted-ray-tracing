84  double intersect(const Ray &r) const
85  {
86  // returns distance, 0 if nohit
87  Vec op = p - r.o; // Solve t^2*d.d + 2*t*(o-p).d + (o-p).(o-p)-R^2 = 0
88  double t, eps = 1e-4,
89  b = op.dot(r.d),
90  det = b *b - op.dot(op) + rad * rad;
91  if (det < 0) return 0;
92  else det = sqrt(det);
93  return (t = b - det) > eps ? t : ((t = b + det) > eps ? t : 0);
94  }

155 Vec radiance(const Ray &r, int depth, unsigned short *Xi, int E = 1)
156 {
157  double t; // distance to intersection
158  int id = 0; // id of intersected object
159  if (!intersect(r, t, id)) return Vec(); // if miss, return black
160  const Sphere &obj = spheres[id]; // the hit object
161  Vec x = r.o + r.d *t, n = (x - obj.p).norm(), nl = n.dot(r.d) < 0 ? n : n *-1, f = obj.c;
162  double p = f.x > f.y && f.x > f.z ? f.x : f.y > f.z ? f.y : f.z; // max refl
163  if (++depth > 5 || !p)
164  if (srand(Xi) < p) {
165  f = f *(1 / p);
166  }
167  else return obj.e * E;
168  if (obj.refl == DIFF)
169  {
170  // Ideal DIFFUSE reflection

172  r2 = srand(Xi),
173  r2s = sqrt(r2);
174  Vec w = nl,
175  u = ((fabs(w.x) > .1 ? Vec(0, 1) : Vec(1)) % w).norm(),
176  v = w % u;
177  Vec d = (u* cos(r1) *r2s + v* sin(r1) *r2s + w* sqrt(1 - r2)).norm();
178 
179  // Loop over any lights
180  Vec e;
181  for (int i = 0; i < numSpheres; i++)
182  {
183  const Sphere &s = spheres[i];
184  if (s.e.x <= 0 && s.e.y <= 0 && s.e.z <= 0) continue; // skip non-lights
185 
186  Vec sw = s.p - x, su = ((fabs(sw.x) > .1 ? Vec(0, 1) : Vec(1)) % sw).norm(), sv = sw % su;
187  double cos_a_max = sqrt(1 - s.rad *s.rad / (x - s.p).dot(x - s.p));
188  double eps1 = srand(Xi), eps2 = srand(Xi);
189  double cos_a = 1 - eps1 + eps1 * cos_a_max;
190  double sin_a = sqrt(1 - cos_a *cos_a);
191  double phi = 2 *M_PI * eps2;
192  Vec l = su* cos(phi) *sin_a + sv* sin(phi) *sin_a + sw * cos_a;
193  l.norm();

195  {
196  // shadow ray
197  double omega = 2 *M_PI *(1 - cos_a_max);
198  e = e + f.mult(s.e *l.dot(nl) *omega) *M_1_PI; // 1/pi for brdf
199  }
200  }
201

203  }

205  return obj.e + f.mult(radiance(Ray(x, r.d - n *2 *n.dot(r.d)), depth, Xi));

208  double nc = 1, nt = 1.5, nnt = into ? nc / nt : nt / nc, ddn = r.d.dot(nl), cos2t;
209  if ((cos2t = 1 - nnt *nnt *(1 - ddn *ddn)) < 0) // Total internal reflection
210  return obj.e + f.mult(radiance(reflRay, depth, Xi));
211  Vec tdir = (r.d *nnt - n *((into ? 1 : -1) *(ddn *nnt + sqrt(cos2t)))).norm();

213  double Re = R0 + (1 - R0) *c *c *c *c *c, Tr = 1 - Re, P = .25 + .5 *Re, RP = Re / P, TP = Tr / (1 - P);
214  return obj.e + f.mult(depth > 2 ? (srand(Xi) < P ? // Russian roulette
215  radiance(reflRay, depth, Xi) *RP : radiance(Ray(x, tdir), depth, Xi) *TP) :
216  radiance(reflRay, depth, Xi) *Re + radiance(Ray(x, tdir), depth, Xi) *Tr);
217 }
